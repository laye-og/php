	{

	#listen: permet de dire c'est quoi le n° de port qui écoute, on peut aussi le changer

	#DocumentRoot:C'est le repertoire de publication.On peut modifier le répertoire par defaut.
	# ServerName : Indique le nom du serveur, par defaut c'est localhost
	# DirectoryIndex: Au cas ou l'utisateur n'indique pas la resource voulue qu'on cherche dans les fichiers  index.php index.php3 index.html index.htm sinon envoi la pge d'accueil
	  
	  httpd-userdir:
	# Userdir: ON l'utilise pour créer des repertoires personnal a chaque programmeur de ce site qui publie des infos par exemple de ce fait dans le site on aura la page de chaque utilisateur.
